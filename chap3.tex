\section{Quotienten von simplizialen Komplexen}
\begin{thLemma}[Inklusion eines Teilkomplexes ist Kofaserung]
    \label{quot:inclusioncofib}
    %
    Sei $K$ ein endlicher simplizialer Komplex und $L\subset K$ ein Teilkomplex.
    Dann ist die Inklusion der Polyeder $\polyeder L \hookrightarrow \polyeder K$
    eine Kofaserung.
\end{thLemma}

\begin{thKorollar}%
    [Quotienten von simplizialen Komplexen nach kontraktiblen Teilkomplexen]
    %
    Sei $K$ ein endlicher simplizialer Komplex und $L\subset K$ ein
    Teilkomplex, für den $\polyeder L$ kontraktibel ist. Dann ist die kanonische
    Projektion $\polyeder K \to \polyeder K\mkern2mu/\mkern2mu\polyeder L$
    eine Homotopieäquivalenz.
\end{thKorollar}
